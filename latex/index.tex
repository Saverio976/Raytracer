\begin{DoxyCode}{0}

\DoxyCodeLine{USAGE: ./raytracer -\/-\/help}
\DoxyCodeLine{}
\DoxyCodeLine{OPTIONS:}
\DoxyCodeLine{    -\/-\/scene-\/path <scene-\/conf.yaax>  path to scene config}
\DoxyCodeLine{    -\/-\/output-\/path <file>    path to output file (dont put .ppm or any extension, it is just a base file path)}
\DoxyCodeLine{    -\/-\/help  to display the help message}
\DoxyCodeLine{    -\/-\/log-\/level <int>   log level can be \{-\/1: no log, 0: fatal, 1: error, 2: warn, 3: info, 4: debug, 5: trace\} [3 by default]}
\DoxyCodeLine{}
